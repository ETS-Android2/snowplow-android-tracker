115  public static long getUTF8Length(@NonNull String s) {
116  long len = 0;
117  for (int i = 0; i < s.length(); i++) {
118  char code = s.charAt(i);
119  if (code <= 0x7f) {
120  len += 1;
121  } else if (code <= 0x7ff) {
122  len += 2;
123  } else if (code >= 0xd800 && code <= 0xdfff) {
124  // Surrogate pair: These take 4 bytes in UTF-8 and 2 chars in UCS-2
125  // (Assume next char is the other [valid] half and just skip it)
126  len += 4; i++;
127  } else if (code < 0xffff) {
128  len += 3;
129  } else {
130  len += 4;
131  }
132  }
133  return len;
134  }

247  @Nullable
248  public static Location getLastKnownLocation(@NonNull Context context) {
249  LocationManager locationManager = (LocationManager) context.getSystemService(Context.LOCATION_SERVICE);
250  Location location = null;
251 
252  try {
253  String locationProvider = null;
254  if (locationManager.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
255  locationProvider = LocationManager.GPS_PROVIDER;
256  } else if (locationManager.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
257  locationProvider = LocationManager.NETWORK_PROVIDER;
258  } else {
259  List<String> locationProviders = locationManager.getProviders(true);
260  if (locationProviders.size() > 0) {
261  locationProvider = locationProviders.get(0);
262  }
263  }
264 
265  if (locationProvider != null && !locationProvider.equals("")) {
266  location = locationManager.getLastKnownLocation(locationProvider);
267  }
268  } catch (SecurityException ex) {
269  Logger.e(TAG, "Exception occurred when retrieving location: %s", ex.toString());
270  }
271 
272  return location;
273  }
