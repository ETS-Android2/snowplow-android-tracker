395  private final NotificationCenter.FunctionalObserver receiveLifecycleNotification = new NotificationCenter.FunctionalObserver() {
396  @Override
397  public void apply(@NonNull Map<String, Object> data) {
398  Session session = getSession();
399  if (lifecycleEvents && session != null) {
400  Boolean isForeground = (Boolean) data.get("isForeground");
401  if (isForeground == null) {
402  return;
403  }
404  int index = session.updateLifecycleNotification(isForeground);
405  // If index is -1 means that the lifecycle is not changed so don't need to send a lifecycle event.
406  if (index == -1) {
407  return;

409  if (isForeground) {

602  public void track(final @NonNull Event event) {
603  if (!dataCollection.get()) {
604  return;
605  }
606  event.beginProcessing(this);
607  TrackerStateSnapshot stateSnapshot;
608  synchronized (this) {
609  stateSnapshot = stateManager.trackerStateForProcessedEvent(event);
610  }
611  boolean reportsOnDiagnostic = !(event instanceof TrackerError);
612  Executor.execute(reportsOnDiagnostic, TAG, () -> {
613  TrackerEvent trackerEvent = new TrackerEvent(event, stateSnapshot);
614  transformEvent(trackerEvent);
615  Payload payload = payloadWithEvent(trackerEvent);
616  Logger.v(TAG, "Adding new payload to event storage: %s", payload);
617  this.emitter.add(payload);
618  event.endProcessing(this);
619  });
620  }

622  private void transformEvent(@NonNull TrackerEvent event) {
623  // Application_install event needs the timestamp to the real installation event.
624  if (event.schema != null
625  && event.schema.equals(TrackerConstants.SCHEMA_APPLICATION_INSTALL)
626  && event.trueTimestamp != null)
627  {
628  event.timestamp = event.trueTimestamp;
629  event.trueTimestamp = null;
630  }
631  // Payload can be optionally updated with values based on internal state
632  stateManager.addPayloadValuesToEvent(event);
633  }

635  private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
636  TrackerPayload payload = new TrackerPayload();
637  addBasicPropertiesToPayload(payload, event);
638  if (event.isPrimitive) {
639  addPrimitivePropertiesToPayload(payload, event);
640  } else {
641  addSelfDescribingPropertiesToPayload(payload, event);
642  }
643  List<SelfDescribingJson> contexts = event.contexts;
644  addBasicContextsToContexts(contexts, event);
645  addGlobalContextsToContexts(contexts, event);
646  addStateMachineEntitiesToContexts(contexts, event);
647  wrapContextsToPayload(payload, contexts);
648  return payload;
649  }

671  private void addSelfDescribingPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
672  payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);
673 
674  workaroundForCampaignAttributionEnrichment(payload, event); // TODO: To remove when Atomic table refactoring is finished
675 
676  SelfDescribingJson data = new SelfDescribingJson(event.schema, event.payload);
677  HashMap<String, Object> unstructuredEventPayload = new HashMap<>();
678  unstructuredEventPayload.put(Parameters.SCHEMA, TrackerConstants.SCHEMA_UNSTRUCT_EVENT);
679  unstructuredEventPayload.put(Parameters.DATA, data.getMap());
680  payload.addMap(unstructuredEventPayload, base64Encoded, Parameters.UNSTRUCTURED_ENCODED, Parameters.UNSTRUCTURED);

683  /*
684  This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
685  the `url` and `referrer` only if they are part of a PageView event.
686  The PageView event is an atomic event but the DeepLinkReceived is a SelfDescribing event.
687  For this reason we copy these two fields in the atomic fields in order to let the enrichment
688  to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
689  This is a hack that should be removed once the atomic event table is dismissed and all the events
690  will be SelfDescribing.
691  */
692  private void workaroundForCampaignAttributionEnrichment(@NonNull Payload payload, @NonNull TrackerEvent event) {
693  if (event.schema.equals(DeepLinkReceived.SCHEMA) && event.payload != null) {
694  String url = (String)event.payload.get(DeepLinkReceived.PARAM_URL);
695  String referrer = (String)event.payload.get(DeepLinkReceived.PARAM_REFERRER);
696  payload.add(Parameters.PAGE_URL, url);
697  payload.add(Parameters.PAGE_REFR, referrer);
698  }
699  }

701  private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {

716  if (trackerSession != null) {
717  synchronized (trackerSession) {
718  SelfDescribingJson sessionContextJson = trackerSession.getSessionContext(eventId);
719  contexts.add(sessionContextJson);

721  } else {
722  Logger.track(TAG, "Session not ready or method getHasLoadedFromFile returned false with eventId: %s", eventId);
723  }
724  }
725 
726  if (geoLocationContext) {
727  contexts.add(Util.getGeoLocationContext(this.context));
728  }
729

731  contexts.add(gdpr.getContext());

806  payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
