112  @SuppressLint("ApplySharedPref")
113  public Session(long foregroundTimeout, long backgroundTimeout, @NonNull TimeUnit timeUnit, @Nullable String namespace, @NonNull Context context) {
114  this.foregroundTimeout = timeUnit.toMillis(foregroundTimeout);
115  this.backgroundTimeout = timeUnit.toMillis(backgroundTimeout);
116  isSessionCheckerEnabled = true;
117 
118  String sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS;
119  if (namespace != null && !namespace.isEmpty()) {
120  String sessionVarsSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
121  sessionVarsName = TrackerConstants.SNOWPLOW_SESSION_VARS + "_" + sessionVarsSuffix;
122  }
123 
124  StrictMode.ThreadPolicy oldPolicy = StrictMode.allowThreadDiskReads();
125  try {
126  sharedPreferences = getSessionFromSharedPreferences(context, sessionVarsName);
127  if (sharedPreferences != null) {
128  userId = sharedPreferences.getString(Parameters.SESSION_USER_ID, Util.getUUIDString());
129  currentSessionId = sharedPreferences.getString(Parameters.SESSION_ID, null);
130  sessionIndex = sharedPreferences.getInt(Parameters.SESSION_INDEX, 0);
131  } else {
132  Map<String, Object> sessionInfo = getSessionFromFile(context);
133  if (sessionInfo != null) {
134  try {
135  userId = sessionInfo.get(Parameters.SESSION_USER_ID).toString();
136  currentSessionId = sessionInfo.get(Parameters.SESSION_ID).toString();
137  sessionIndex = (int) sessionInfo.get(Parameters.SESSION_INDEX);
138  } catch (Exception e) {
139  Logger.track(TAG, String.format("Exception occurred retrieving session info from file: %s", e), e);
140  userId = Util.getUUIDString();
141  }
142  } else {
143  userId = Util.getUUIDString();
144  }
145  }
146  // Force sharedPreferences to be the correct one.
147  sharedPreferences = context.getSharedPreferences(TrackerConstants.SNOWPLOW_SESSION_VARS, Context.MODE_PRIVATE);
148  lastSessionCheck = System.currentTimeMillis();
149  } finally {
150  StrictMode.setThreadPolicy(oldPolicy);
151  }
152 
153  // Get or Set the Session UserID
154  SharedPreferences generalPref = context.getSharedPreferences(TrackerConstants.SNOWPLOW_GENERAL_VARS, Context.MODE_PRIVATE);
155  String storedUserId = generalPref.getString(TrackerConstants.INSTALLATION_USER_ID, null);
156  if (storedUserId != null) {
157  userId = storedUserId;
158  } else if (userId != null) {
159  generalPref.edit()
160  .putString(TrackerConstants.INSTALLATION_USER_ID, userId)
161  .commit();
162  }
163 
164  Logger.v(TAG, "Tracker Session Object created.");
165  }

196  private boolean shouldUpdateSession() {
197  if (isNewSession.get()) {
198  return true;
199  }
200  long now = System.currentTimeMillis();
201  long timeout = isBackground.get() ? backgroundTimeout : foregroundTimeout;
202  return now < lastSessionCheck || now - lastSessionCheck > timeout;
203  }

205  private synchronized void updateSession(String eventId) {
206  isNewSession.set(false);
207  firstId = eventId;
208  previousSessionId = this.currentSessionId;
209  currentSessionId = Util.getUUIDString();
210  sessionIndex++;
211 
212  Logger.d(TAG, "Session information is updated:");
213  Logger.d(TAG, " + Session ID: %s", currentSessionId);
214  Logger.d(TAG, " + Previous Session ID: %s", previousSessionId);
215  Logger.d(TAG, " + Session Index: %s", sessionIndex);
216 
217  boolean isBackground = this.isBackground.get();
218 
219  if (isBackground) { // timed out in background
220  this.executeEventCallback(backgroundTimeoutCallback);
221  } else { // timed out in foreground
222  this.executeEventCallback(foregroundTimeoutCallback);
223  }
224 
225  SharedPreferences.Editor editor = sharedPreferences.edit();
226  editor.putString(Parameters.SESSION_ID, currentSessionId);
227  editor.putString(Parameters.SESSION_PREVIOUS_ID, previousSessionId);
228  editor.putInt(Parameters.SESSION_INDEX, sessionIndex);
229  editor.putString(Parameters.SESSION_FIRST_ID, firstId);
230  editor.putString(Parameters.SESSION_STORAGE, sessionStorage);
231  editor.apply();
232  }

244  public synchronized int updateLifecycleNotification(boolean isForeground) {
245  boolean toBackground = !isForeground;
246  // if the new lifecycle state confirms the session state, there isn't any lifecycle transition
247  if (isBackground.get() == toBackground) {
248  return -1;
249  }
250  Logger.d(TAG, "Application is in the background: %s", toBackground);
251  isBackground.set(toBackground);
252 
253  if (!toBackground) {
254  Logger.d(TAG, "Application moved to foreground, starting session checking...");
255  this.executeEventCallback(this.foregroundTransitionCallback);
256  try {
257  setIsSuspended(false);
258  } catch (Exception e) {
259  Logger.e(TAG, "Could not resume checking as tracker not setup. Exception: %s", e);
260  }
261  foregroundIndex++;
262  return foregroundIndex;
263  } else {
264  Logger.d(TAG, "Application moved to background");
265  this.executeEventCallback(this.backgroundTransitionCallback);
266  backgroundIndex++;
267  return backgroundIndex;
268  }
269  }
