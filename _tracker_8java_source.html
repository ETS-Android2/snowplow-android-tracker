521  public void track(final Event event) {
522  if (!dataCollection.get()) {
523  return;
524  }
525 
526  if (event instanceof ScreenView && screenState != null) {
527  ScreenView screenView = (ScreenView) event;
528  screenView.updateScreenState(screenState);
529  }
530 
531  boolean reportsOnDiagnostic = !(event instanceof TrackerError);
532  Executor.execute(reportsOnDiagnostic, TAG, () -> {
533  event.beginProcessing(this);
534  processEvent(event);
535  event.endProcessing(this);
536  });

539  private void processEvent(@NonNull Event event) {
540  TrackerEvent trackerEvent = new TrackerEvent(event);

546  private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
547  TrackerPayload payload = new TrackerPayload();
548  addBasicPropertiesToPayload(payload, event);
549  if (event.isPrimitive) {
550  addPrimitivePropertiesToPayload(payload, event);
551  } else {
552  addSelfDescribingPropertiesToPayload(payload, event);
553  }
554  List<SelfDescribingJson> contexts = event.contexts;
555  addBasicContextsToContexts(contexts, event);
556  addGlobalContextsToContexts(contexts, payload);
557  wrapContextsToPayload(payload, contexts);
558  return payload;
559  }

561  private void addBasicPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
562  payload.add(Parameters.EID, event.eventId.toString());
563  payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));
564  if (event.trueTimestamp != null) {
565  payload.add(Parameters.TRUE_TIMESTAMP, event.trueTimestamp.toString());
566  }
567  payload.add(Parameters.APPID, this.appId);
568  payload.add(Parameters.NAMESPACE, this.namespace);
569  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
570  if (this.subject != null) {
571  payload.addMap(new HashMap<>(this.subject.getSubject()));
572  }
573  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
574  }

579  }

581  private void addSelfDescribingPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
582  payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);
583  SelfDescribingJson data = new SelfDescribingJson(event.schema, event.payload);
584  HashMap<String, Object> unstructuredEventPayload = new HashMap<>();
585  unstructuredEventPayload.put(Parameters.SCHEMA, TrackerConstants.SCHEMA_UNSTRUCT_EVENT);
586  unstructuredEventPayload.put(Parameters.DATA, data.getMap());
587  payload.addMap(unstructuredEventPayload, base64Encoded, Parameters.UNSTRUCTURED_ENCODED, Parameters.UNSTRUCTURED);
588  }

590  private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {
591  if (applicationContext) {
592  contexts.add(InstallTracker.getApplicationContext(this.context));
593  }
594 
595  if (mobileContext) {
596  contexts.add(Util.getMobileContext(this.context));
597  }
598 
599  if (event.isService) {
600  return;
601  }
602 
603  if (sessionContext) {
604  String eventId = event.eventId.toString();
605  if (trackerSession.getHasLoadedFromFile()) {
606  synchronized (trackerSession) {
607  SelfDescribingJson sessionContextJson = trackerSession.getSessionContext(eventId);
608  if (sessionContextJson == null) {
609  Logger.track(TAG, "Method getSessionContext method returned null with eventId: %s", eventId);
610  }
611  contexts.add(sessionContextJson);
612  }
613  } else {
614  Logger.track(TAG, "Method getHasLoadedFromFile method returned false with eventId: %s", eventId);
615  }
616  }
617 
618  if (geoLocationContext) {
619  contexts.add(Util.getGeoLocationContext(this.context));
620  }
621 
622  if (screenContext) {
623  contexts.add(screenState.getCurrentScreen(true));
624  }
625 
626  if (gdpr != null) {
627  contexts.add(gdpr.getContext());
628  }
629  }

639  private void wrapContextsToPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
640  if (contexts.isEmpty()) {
641  return;
642  }
643  List<Map> data = new LinkedList<>();
644  for (SelfDescribingJson context : contexts) {
645  if (context != null) {
646  data.add(context.getMap());
647  }
648  }
649  SelfDescribingJson finalContext = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, data);
650  if (finalContext == null) {
651  return;
652  }
653  payload.addMap(finalContext.getMap(), base64Encoded, Parameters.CONTEXT_ENCODED, Parameters.CONTEXT);
654  }

669  private void addServiceEventPayload(Payload payload, List<SelfDescribingJson> contexts) {
670  // Add default parameters to the payload
671  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
672  payload.add(Parameters.APPID, this.appId);
673  payload.add(Parameters.NAMESPACE, this.namespace);
674  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
675 
676  // If there is a subject present for the Tracker add it
677  if (this.subject != null) {
678  payload.addMap(new HashMap<>(this.subject.getSubject()));
679  }
680 
681  // Add Mobile Context
682  if (this.mobileContext) {
683  contexts.add(Util.getMobileContext(this.context));
684  }
685 
686  // Add application context
687  if (this.applicationContext) {
688  contexts.add(InstallTracker.getApplicationContext(this.context));
689  }
690 
691  // If there are contexts to nest
692  if (contexts.size() > 0) {
693  List<Map> contextMaps = new LinkedList<>();
694  for (SelfDescribingJson selfDescribingJson : contexts) {
695  if (selfDescribingJson != null) {
696  contextMaps.add(selfDescribingJson.getMap());
697  }
698  }
699  SelfDescribingJson envelope = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, contextMaps);
700  payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
701  Parameters.CONTEXT);
702  }
703 
704  // Add this payload to the emitter
705  this.emitter.add(payload);
706  }

736  public void resumeSessionChecking() {
737  if (sessionExecutor == null && this.sessionContext) {
738  Logger.d(TAG, "Session checking has been resumed.");
739  final Session session = this.trackerSession;
740  sessionExecutor = Executors.newSingleThreadScheduledExecutor();
741  sessionExecutor.scheduleAtFixedRate(new Runnable() {
742  @Override
743  public void run() {
744  try {
745  session.checkAndUpdateSession();
746  } catch (Exception e) {
747  Logger.track(TAG, "Method checkAndUpdateSession raised an exception: %s", e);
748  }
749  }
750  }, this.sessionCheckInterval, this.sessionCheckInterval, this.timeUnit);
751  }
752  }
