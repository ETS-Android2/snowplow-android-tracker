638  public void track(final @NonNull Event event) {
639  if (!dataCollection.get()) {
640  return;
641  }
642

664  private void transformEvent(@NonNull TrackerEvent event) {
665  // Application_install event needs the timestamp to the real installation event.
666  if (event.schema != null
667  && event.schema.equals(TrackerConstants.SCHEMA_APPLICATION_INSTALL)
668  && event.trueTimestamp != null)

670  event.timestamp = event.trueTimestamp;
671  event.trueTimestamp = null;
672  }

675  private @NonNull Payload payloadWithEvent(@NonNull TrackerEvent event) {
676  TrackerPayload payload = new TrackerPayload();
677  addBasicPropertiesToPayload(payload, event);
678  if (event.isPrimitive) {
679  addPrimitivePropertiesToPayload(payload, event);
680  } else {
681  addSelfDescribingPropertiesToPayload(payload, event);
682  }
683  List<SelfDescribingJson> contexts = event.contexts;
684  addBasicContextsToContexts(contexts, event);
685  addGlobalContextsToContexts(contexts, event);
686  wrapContextsToPayload(payload, contexts);
687  return payload;
688  }

690  private void addBasicPropertiesToPayload(@NonNull Payload payload, @NonNull TrackerEvent event) {
691  payload.add(Parameters.EID, event.eventId.toString());
692  payload.add(Parameters.DEVICE_TIMESTAMP, Long.toString(event.timestamp));
693  if (event.trueTimestamp != null) {
694  payload.add(Parameters.TRUE_TIMESTAMP, event.trueTimestamp.toString());
695  }
696  payload.add(Parameters.APPID, this.appId);
697  payload.add(Parameters.NAMESPACE, this.namespace);
698  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
699  if (this.subject != null) {
700  payload.addMap(new HashMap<>(this.subject.getSubject()));
701  }
702  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
703  }

711  payload.add(Parameters.EVENT, TrackerConstants.EVENT_UNSTRUCTURED);

719  private void addBasicContextsToContexts(@NonNull List<SelfDescribingJson> contexts, @NonNull TrackerEvent event) {
720  if (applicationContext) {
721  contexts.add(Util.getApplicationContext(this.context));
722  }
723 
724  if (mobileContext) {
725  contexts.add(Util.getMobileContext(this.context));
726  }
727 
728  if (event.isService) {
729  return;
730  }
731 
732  if (sessionContext) {
733  String eventId = event.eventId.toString();
734  if (trackerSession != null) {
735  synchronized (trackerSession) {
736  SelfDescribingJson sessionContextJson = trackerSession.getSessionContext(eventId);
737  if (sessionContextJson == null) {
738  Logger.track(TAG, "Method getSessionContext returned null with eventId: %s", eventId);
739  }
740  contexts.add(sessionContextJson);
741  }
742  } else {
743  Logger.track(TAG, "Session not ready or method getHasLoadedFromFile returned false with eventId: %s", eventId);
744  }
745  }
746 
747  if (geoLocationContext) {
748  contexts.add(Util.getGeoLocationContext(this.context));
749  }
750 
751  if (screenContext) {
752  contexts.add(screenState.getCurrentScreen(true));
753  }
754 
755  if (gdpr != null) {
756  contexts.add(gdpr.getContext());
757  }
758  }

768  private void wrapContextsToPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
769  if (contexts.isEmpty()) {
770  return;
771  }
772  List<Map> data = new LinkedList<>();
773  for (SelfDescribingJson context : contexts) {
774  if (context != null) {
775  data.add(context.getMap());
776  }
777  }
778  SelfDescribingJson finalContext = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, data);
779  if (finalContext == null) {
780  return;
781  }
782  payload.addMap(finalContext.getMap(), base64Encoded, Parameters.CONTEXT_ENCODED, Parameters.CONTEXT);
783  }

798  private void addServiceEventPayload(@NonNull Payload payload, @NonNull List<SelfDescribingJson> contexts) {
799  // Add default parameters to the payload
800  payload.add(Parameters.PLATFORM, this.devicePlatform.getValue());
801  payload.add(Parameters.APPID, this.appId);
802  payload.add(Parameters.NAMESPACE, this.namespace);
803  payload.add(Parameters.TRACKER_VERSION, this.trackerVersion);
804 
805  // If there is a subject present for the Tracker add it
806  if (this.subject != null) {
807  payload.addMap(new HashMap<>(this.subject.getSubject()));
808  }
809 
810  // Add Mobile Context
811  if (this.mobileContext) {
812  contexts.add(Util.getMobileContext(this.context));
813  }
814 
815  // Add application context
816  if (this.applicationContext) {
817  contexts.add(Util.getApplicationContext(this.context));
818  }
819 
820  // If there are contexts to nest
821  if (contexts.size() > 0) {
822  List<Map> contextMaps = new LinkedList<>();
823  for (SelfDescribingJson selfDescribingJson : contexts) {
824  if (selfDescribingJson != null) {
825  contextMaps.add(selfDescribingJson.getMap());
826  }
827  }
828  SelfDescribingJson envelope = new SelfDescribingJson(TrackerConstants.SCHEMA_CONTEXTS, contextMaps);
829  payload.addMap(envelope.getMap(), this.base64Encoded, Parameters.CONTEXT_ENCODED,
830  Parameters.CONTEXT);
831  }
832

834  this.emitter.add(payload);
