71  @NonNull
72  public synchronized static List<String> removeUnsentEventsExceptForNamespaces(@NonNull Context context, @Nullable List<String> allowedNamespaces) {
73  String[] databaseList = context.databaseList();
74  if (databaseList == null) {
75  return new ArrayList<>();
76  }
77  List<String> allowedDbFiles = new ArrayList<>(allowedNamespaces.size());
78  for (String namespace : allowedNamespaces) {
79  String sqliteSuffix = namespace.replaceAll("[^a-zA-Z0-9_]+", "-");
80  String dbName = DATABASE_NAME + "-" + sqliteSuffix + ".sqlite";
81  allowedDbFiles.add(dbName);
82  allowedDbFiles.add(dbName + "-wal");
83  allowedDbFiles.add(dbName + "-shm");
84  }
85  List<String> removedDbFiles = new ArrayList<>();
86  for (String dbName : databaseList) {
87  if (!dbName.startsWith(DATABASE_NAME)) {
88  continue;
89  }
90  if (!allowedDbFiles.contains(dbName)) {
91  if (context.deleteDatabase(dbName)) {
92  removedDbFiles.add(dbName);
93  }
94  }
95  }

97  }
