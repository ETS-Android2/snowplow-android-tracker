477  @SuppressWarnings("all")
478  private void attemptEmit() {
479  if (!Util.isOnline(this.context)) {
480  Logger.d(TAG, "Emitter loop stopping: emitter offline.");
481  isRunning.compareAndSet(true, false);
482  return;
483  }
484  if (eventStore.getSize() <= 0) {
485  if (emptyCount >= this.emptyLimit) {
486  Logger.d(TAG, "Emitter loop stopping: empty limit reached.");
487  isRunning.compareAndSet(true, false);
488  return;
489  }
490  emptyCount++;
491  Logger.e(TAG, "Emitter database empty: " + emptyCount);
492  try {
493  this.timeUnit.sleep(this.emitterTick);
494  } catch (InterruptedException e) {
495  Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
496  }
497  attemptEmit();
498  return;
499  }
500  emptyCount = 0;
501 
502  List<EmitterEvent> events = eventStore.getEmittableEvents(sendLimit);
503  List<Request> requests = buildRequests(events);
504  List<RequestResult> results = networkConnection.sendRequests(requests);
505 
506  Logger.v(TAG, "Processing emitter results.");
507 
508  int successCount = 0;
509  int failureCount = 0;
510  List<Long> removableEvents = new ArrayList<>();
511 
512  for (RequestResult res : results) {
513  if (res.getSuccess()) {
514  removableEvents.addAll(res.getEventIds());
515  successCount += res.getEventIds().size();
516  } else {
517  failureCount += res.getEventIds().size();
518  Logger.e(TAG, "Request sending failed but we will retry later.");
519  }
520  }
521  eventStore.removeEvents(removableEvents);
522 
523  Logger.d(TAG, "Success Count: %s", successCount);
524  Logger.d(TAG, "Failure Count: %s", failureCount);
525 
526  if (requestCallback != null) {
527  if (failureCount != 0) {
528  requestCallback.onFailure(successCount, failureCount);
529  } else {
530  requestCallback.onSuccess(successCount);
531  }
532  }
533 
534  if (failureCount > 0 && successCount == 0) {
535  if (Util.isOnline(this.context)) {
536  Logger.e(TAG, "Ensure collector path is valid: %s", getEmitterUri());
537  }
538  Logger.e(TAG, "Emitter loop stopping: failures.");
539  isRunning.compareAndSet(true, false);
540  } else {
541  attemptEmit();
542  }

555  @NonNull
556  protected List<Request> buildRequests(@NonNull List<EmitterEvent> events) {
557  List<Request> requests = new ArrayList<>();
558  String sendingTime = Util.getTimestamp();
559  HttpMethod httpMethod = networkConnection.getHttpMethod();
560 
561  if (httpMethod == GET) {
562  for (EmitterEvent event : events) {
563  Payload payload = event.payload;
564  addSendingTimeToPayload(payload, sendingTime);
565  boolean isOversize = isOversize(payload);
566  Request request = new Request(payload, event.eventId, isOversize);
567  requests.add(request);
568  }
569  } else {
570  for (int i = 0; i < events.size(); i += bufferOption.getCode()) {
571  List<Long> reqEventIds = new ArrayList<>();
572  List<Payload> postPayloadMaps = new ArrayList<>();
573 
574  for (int j = i; j < (i + bufferOption.getCode()) && j < events.size(); j++) {
575  EmitterEvent event = events.get(j);
576  Payload payload = event.payload;
577  Long eventId = event.eventId;
578  addSendingTimeToPayload(payload, sendingTime);
579 
580  if (isOversize(payload)) {
581  Request request = new Request(payload, eventId, true);
582  requests.add(request);
583 
584  } else if (isOversize(payload, postPayloadMaps)) {
585  Request request = new Request(postPayloadMaps, reqEventIds);
586  requests.add(request);
587 
588  // Clear collections and build a new POST
589  postPayloadMaps = new ArrayList<>();
590  reqEventIds = new ArrayList<>();
591 
592  // Build and store the request
593  postPayloadMaps.add(payload);
594  reqEventIds.add(eventId);
595 
596  } else {
597  postPayloadMaps.add(payload);
598  reqEventIds.add(eventId);
599  }
600  }
601 
602  // Check if all payloads have been processed
603  if (!postPayloadMaps.isEmpty()) {
604  Request request = new Request(postPayloadMaps, reqEventIds);
605  requests.add(request);
606  }
607  }
608  }
609  return requests;
610  }

639  private boolean isOversize(@NonNull Payload payload, long byteLimit, @NonNull List<Payload> previousPaylods) {
640  long totalByteSize = payload.getByteSize();
641  for (Payload previousPayload : previousPaylods) {
642  totalByteSize += previousPayload.getByteSize();
643  }
644  int wrapperBytes = previousPaylods.size() > 0 ? (previousPaylods.size() + POST_WRAPPER_BYTES) : 0;
645  return totalByteSize + wrapperBytes > byteLimit;
646  }
