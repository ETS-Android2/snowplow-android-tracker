57  public boolean matchWithSchema(@NonNull String schema) {
58  if (schema == null) {
59  return false;
60  }
61  List<String> uriParts = getParts(schema, URI_PATTERN);
62  if (uriParts == null || uriParts.size() < ruleParts.size()) {
63  return false;
64  }
65  // Check vendor part
66  String[] ruleVendor = ruleParts.get(0).split("\\.");
67  String[] uriVendor = uriParts.get(0).split("\\.");
68  if (uriVendor.length != ruleVendor.length) {
69  return false;
70  }
71  int index = 0;
72  for (String ruleVendorPart : ruleVendor) {
73  if (!"*".equals(ruleVendorPart) && !uriVendor[index].equals(ruleVendorPart)) {
74  return false;
75  }
76  index++;
77  }
78  // Check the rest of the rule
79  index = 1;
80  for (String rulePart : ruleParts.subList(1, ruleParts.size())) {
81  if (!"*".equals(rulePart) && !uriParts.get(index).equals(rulePart)) {
82  return false;
83  }
84  index++;
85  }
86  return true;

107  private static boolean validateVendor(@NonNull String vendor) {
108  // the components array will be generated like this from vendor:
109  // "com.acme.marketing" => ["com", "acme", "marketing"]
110  String[] components = vendor.split("\\.");
111  // check that vendor doesn't begin or end with period
112  // e.g. ".snowplowanalytics.snowplow." => ["", "snowplowanalytics", "snowplow", ""]
113  if (components.length > 1 && (components[0].isEmpty() || components[components.length-1].isEmpty())) {
114  return false;
115  }
116  // reject vendors with criteria that are too broad & don't make sense, i.e. "*.*.marketing"
117  if ("*".equals(components[0]) || "*".equals(components[1])) {
118  return false;
119  }
120  // now validate the remaining parts, vendors should follow matching that never breaks trailing specificity
121  // in other words, once we use an asterisk, we must continue using asterisks for parts or stop
122  // e.g. "com.acme.marketing.*.*" is allowed, but "com.acme.*.marketing.*" or "com.acme.*.marketing" is forbidden
123  if (components.length <= 2) return true;
124  // trailingComponents are the remaining parts after the first two
125  String[] trailingComponents = Arrays.copyOfRange(components, 2, components.length);
126  boolean asterisk = false;
127  for (String part : trailingComponents) {
128  if ("*".equals(part)) { // mark when we've found a wildcard
129  asterisk = true;
130  } else if (asterisk) { // invalid when alpha parts come after wildcard
131  return false;
132  }
133  }
134  return true;
135  }
