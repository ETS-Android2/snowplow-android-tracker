177  private List<Map<String, Object>> queryDatabase(String query, String orderBy) {
178  List<Map<String, Object>> res = new ArrayList<>();
179  if (isDatabaseOpen()) {
180  Cursor cursor = null;
181  try {
182  cursor = database.query(EventStoreHelper.TABLE_EVENTS, allColumns, query, null, null, null, orderBy);
183  cursor.moveToFirst();
184  while (!cursor.isAfterLast()) {
185  Map<String, Object> eventMetadata = new HashMap<>();
186  eventMetadata.put(EventStoreHelper.METADATA_ID, cursor.getLong(0));
187  eventMetadata.put(EventStoreHelper.METADATA_EVENT_DATA,
188  Util.deserializer(cursor.getBlob(1)));
189  eventMetadata.put(EventStoreHelper.METADATA_DATE_CREATED, cursor.getString(2));
190  cursor.moveToNext();
191  res.add(eventMetadata);
192  }
193  } finally {
194  if (cursor != null) {
195  cursor.close();
196  }
197  }
198  }
199  return res;
200  }

236  LinkedList<Long> eventIds = new LinkedList<>();
237  ArrayList<Payload> events = new ArrayList<>();
238 
239  // FIFO Pattern for sending events
240  for (Map<String, Object> eventMetadata : getDescEventsInRange(this.sendLimit)) {
241 
242  // Create a TrackerPayload for each event
243  TrackerPayload payload = new TrackerPayload();
244  Map<String, Object> eventData = (Map<String, Object>)
245  eventMetadata.get(EventStoreHelper.METADATA_EVENT_DATA);
246  payload.addMap(eventData);
247 
248  // Store the eventId
249  Long eventId = (Long) eventMetadata.get(EventStoreHelper.METADATA_ID);
250  eventIds.add(eventId);
251 
252  // Add the payload to the list
253  events.add(payload);
254  }
255  return new EmittableEvents(events, eventIds);
256  }
