297 public Emitter(@NonNull Context context, @NonNull String collectorUri, @Nullable EmitterBuilder builder) {
298 this.context = context;
299 if (builder == null) {
300 builder = new EmitterBuilder();
301 }
302 this.requestCallback = builder.requestCallback;
303 this.bufferOption = builder.bufferOption;
304 this.requestSecurity = builder.requestSecurity;
305 this.tlsVersions = builder.tlsVersions;
306 this.emitterTick = builder.emitterTick;
307 this.emptyLimit = builder.emptyLimit;
308 this.sendLimit = builder.sendLimit;
309 this.byteLimitGet = builder.byteLimitGet;
310 this.byteLimitPost = builder.byteLimitPost;
311 this.emitTimeout = builder.emitTimeout;
312 this.timeUnit = builder.timeUnit;
313 this.client = builder.client;
314 this.eventStore = builder.eventStore;
315
316 this.uri = collectorUri;
317 this.httpMethod = builder.httpMethod;
318 this.customPostPath = builder.customPostPath;
319 if (builder.networkConnection == null) {
320 isCustomNetworkConnection = false;
321 String endpoint = collectorUri;
322 if (!endpoint.startsWith("http")) {
323 String protocol = builder.requestSecurity == Protocol.HTTPS ? "https://" : "http://";
324 endpoint = protocol + endpoint;
325 }
326 this.uri = endpoint;
327 setNetworkConnection(new OkHttpNetworkConnection.OkHttpNetworkConnectionBuilder(endpoint)
328 .method(builder.httpMethod)
329 .tls(builder.tlsVersions)
330 .emitTimeout(builder.emitTimeout)
331 .customPostPath(builder.customPostPath)
332 .client(builder.client)
333 .build());
334 } else {
335 isCustomNetworkConnection = true;
336 setNetworkConnection(builder.networkConnection);
337 }
338
339 if (builder.threadPoolSize > 2) {
340 Executor.setThreadCount(builder.threadPoolSize);
341 }
342
343 Logger.v(TAG, "Emitter created successfully!");
344 }

415 public boolean shutdown(long timeout) {
416 Logger.d(TAG, "Shutting down emitter.");
417 isRunning.compareAndSet(true, false);
418 ExecutorService es = Executor.shutdown();
419 if (es == null || timeout <= 0) {
420 return true;
421 }
422 try {
423 boolean isTerminated = es.awaitTermination(timeout, TimeUnit.SECONDS);
424 Logger.d(TAG, "Executor is terminated: " + isTerminated);
425 return isTerminated;
426 } catch (InterruptedException e) {
427 Logger.e(TAG, "Executor termination is interrupted: " + e.getMessage());
428 return false;
429 }

447 @SuppressWarnings("all")
448 private void attemptEmit(NetworkConnection networkConnection) {
449 if (isEmittingPaused.get()) {
450 Logger.d(TAG, "Emitter paused.");
451 isRunning.compareAndSet(true, false);
452 return;
453 }
454 if (!Util.isOnline(this.context)) {
455 Logger.d(TAG, "Emitter loop stopping: emitter offline.");
456 isRunning.compareAndSet(true, false);
457 return;
458 }
459 if (eventStore.getSize() <= 0) {
460 if (emptyCount >= this.emptyLimit) {
461 Logger.d(TAG, "Emitter loop stopping: empty limit reached.");
462 isRunning.compareAndSet(true, false);
463 return;
464 }
465 emptyCount++;
466 Logger.e(TAG, "Emitter database empty: " + emptyCount);
467 try {
468 this.timeUnit.sleep(this.emitterTick);
469 } catch (InterruptedException e) {
470 Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
471 }
472 attemptEmit(getNetworkConnection()); // at this point we update network connection since it might be outdated after sleep
473 return;
474 }
475 emptyCount = 0;
476
477 List<EmitterEvent> events = eventStore.getEmittableEvents(sendLimit);
478 List<Request> requests = buildRequests(events, networkConnection.getHttpMethod());
479 List<RequestResult> results = networkConnection.sendRequests(requests);
480
481 Logger.v(TAG, "Processing emitter results.");
482
483 int successCount = 0;
484 int failureCount = 0;
485 List<Long> removableEvents = new ArrayList<>();
486
487 for (RequestResult res : results) {
488 if (res.getSuccess()) {
489 removableEvents.addAll(res.getEventIds());
490 successCount += res.getEventIds().size();
491 } else {
492 failureCount += res.getEventIds().size();
493 Logger.e(TAG, "Request sending failed but we will retry later.");
494 }
495 }
496 eventStore.removeEvents(removableEvents);
497
498 Logger.d(TAG, "Success Count: %s", successCount);
499 Logger.d(TAG, "Failure Count: %s", failureCount);
500
501 if (requestCallback != null) {
502 if (failureCount != 0) {
503 requestCallback.onFailure(successCount, failureCount);
504 } else {
505 requestCallback.onSuccess(successCount);
506 }
507 }
508
509 if (failureCount > 0 && successCount == 0) {
510 if (Util.isOnline(this.context)) {
511 Logger.e(TAG, "Ensure collector path is valid: %s", networkConnection.getUri());
512 }
513 Logger.e(TAG, "Emitter loop stopping: failures.");
514 isRunning.compareAndSet(true, false);
515 } else {
516 attemptEmit(getNetworkConnection()); // refresh network connection for next emit
517 }
518 }

531 @NonNull
532 protected List<Request> buildRequests(@NonNull List<EmitterEvent> events, HttpMethod httpMethod) {
533 List<Request> requests = new ArrayList<>();
534 String sendingTime = Util.getTimestamp();
535
536 if (httpMethod == GET) {
537 for (EmitterEvent event : events) {
538 Payload payload = event.payload;
539 addSendingTimeToPayload(payload, sendingTime);
540 boolean isOversize = isOversize(payload, httpMethod);
541 Request request = new Request(payload, event.eventId, isOversize);
542 requests.add(request);
543 }
544 } else {
545 for (int i = 0; i < events.size(); i += bufferOption.getCode()) {
546 List<Long> reqEventIds = new ArrayList<>();
547 List<Payload> postPayloadMaps = new ArrayList<>();
548
549 for (int j = i; j < (i + bufferOption.getCode()) && j < events.size(); j++) {
550 EmitterEvent event = events.get(j);
551 Payload payload = event.payload;
552 Long eventId = event.eventId;
553 addSendingTimeToPayload(payload, sendingTime);
554
555 if (isOversize(payload, httpMethod)) {
556 Request request = new Request(payload, eventId, true);
557 requests.add(request);
558
559 } else if (isOversize(payload, postPayloadMaps, httpMethod)) {
560 Request request = new Request(postPayloadMaps, reqEventIds);
561 requests.add(request);
562
563 // Clear collections and build a new POST
564 postPayloadMaps = new ArrayList<>();
565 reqEventIds = new ArrayList<>();
566
567 // Build and store the request
568 postPayloadMaps.add(payload);
569 reqEventIds.add(eventId);
570
571 } else {
572 postPayloadMaps.add(payload);
573 reqEventIds.add(eventId);
574 }
575 }
576
577 // Check if all payloads have been processed
578 if (!postPayloadMaps.isEmpty()) {
579 Request request = new Request(postPayloadMaps, reqEventIds);
580 requests.add(request);
581 }
582 }
583 }
584 return requests;
585 }

604 private boolean isOversize(@NonNull Payload payload, @NonNull List<Payload> previousPaylods, HttpMethod httpMethod) {
605 long byteLimit = httpMethod == GET ? byteLimitGet : byteLimitPost;
606 return isOversize(payload, byteLimit, previousPaylods);
607 }
608
616 private boolean isOversize(@NonNull Payload payload, long byteLimit, @NonNull List<Payload> previousPaylods) {
617 long totalByteSize = payload.getByteSize();
618 for (Payload previousPayload : previousPaylods) {
619 totalByteSize += previousPayload.getByteSize();
620 }
621 int wrapperBytes = previousPaylods.size() > 0 ? (previousPaylods.size() + POST_WRAPPER_BYTES) : 0;
622 return totalByteSize + wrapperBytes > byteLimit;
623 }
