167  @Nullable
168  public Pair<String, Integer> getBatteryStateAndLevel(@NonNull Context context) {
169  IntentFilter ifilter = new IntentFilter(Intent.ACTION_BATTERY_CHANGED);
170  Intent batteryIntent = context.registerReceiver(null, ifilter);
171  if (batteryIntent == null) {
172  return null;
173  }
174 
175  int status = batteryIntent.getIntExtra(BatteryManager.EXTRA_STATUS, -1);
176  int level = batteryIntent.getIntExtra(BatteryManager.EXTRA_LEVEL, -1);
177  int scale = batteryIntent.getIntExtra(BatteryManager.EXTRA_SCALE, -1);
178  if (status == -1 || level == -1 || scale == -1) {
179  return null;
180  }
181 
182  String batteryState;
183  switch (status) {
184  case BatteryManager.BATTERY_HEALTH_UNKNOWN:
185  batteryState = null;
186  break;
187  case BatteryManager.BATTERY_STATUS_FULL:
188  batteryState = "full";
189  break;
190  case BatteryManager.BATTERY_STATUS_CHARGING:
191  batteryState = "charging";
192  break;
193  default:
194  batteryState = "unplugged";
195  break;
196  }
197 
198  int batteryLevel = (int) (level * 100 / (float)scale);
199 
200  return new Pair<>(batteryState, batteryLevel);
201  }
