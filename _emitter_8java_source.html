425  @SuppressWarnings("all")
426  private void attemptEmit() {
427  if (eventStore != null) {
428  if (Util.isOnline(this.context)) {
429  if (eventStore.getSize() > 0) {
430  emptyCount = 0;
431 
432  EmittableEvents events = eventStore.getEmittableEvents();
433  LinkedList<ReadyRequest> requests = buildRequests(events);
434  LinkedList<RequestResult> results = performAsyncEmit(requests);
435 
436  events = null;
437  requests = null;
438 
439  Logger.v(TAG, "Processing emitter results.");
440 
441  int successCount = 0;
442  int failureCount = 0;
443  LinkedList<Long> removableEvents = new LinkedList<>();
444 
445  for (RequestResult res : results) {
446  if (res.getSuccess()) {
447  for (final Long eventId : res.getEventIds()) {
448  removableEvents.add(eventId);
449  }
450  successCount += res.getEventIds().size();
451  } else {
452  failureCount += res.getEventIds().size();
453  Logger.e(TAG, "Request sending failed but we will retry later.");
454  }
455  }
456  eventStore.removeEvents(removableEvents);
457 
458  results = null;
459  removableEvents = null;
460 
461  Logger.d(TAG, "Success Count: %s", successCount);
462  Logger.d(TAG, "Failure Count: %s", failureCount);
463 
464  if (requestCallback != null) {
465  if (failureCount != 0) {
466  requestCallback.onFailure(successCount, failureCount);
467  } else {
468  requestCallback.onSuccess(successCount);
469  }
470  }
471 
472  if (failureCount > 0 && successCount == 0) {
473  if (Util.isOnline(this.context)) {
474  Logger.e(TAG, "Ensure collector path is valid: %s", getEmitterUri());
475  }
476  Logger.e(TAG, "Emitter loop stopping: failures.");
477  isRunning.compareAndSet(true, false);
478  } else {
479  attemptEmit();
480  }
481  } else {
482  if (emptyCount >= this.emptyLimit) {
483  Logger.e(TAG, "Emitter loop stopping: empty limit reached.");
484  isRunning.compareAndSet(true, false);
485  } else {
486  emptyCount++;
487  Logger.e(TAG, "Emitter database empty: " + emptyCount);
488  try {
489  this.timeUnit.sleep(this.emitterTick);
490  } catch (InterruptedException e) {
491  Logger.e(TAG, "Emitter thread sleep interrupted: " + e.toString());
492  }
493  attemptEmit();
494  }
495  }
496  } else {
497  Logger.e(TAG, "Emitter loop stopping: emitter offline.");
498  isRunning.compareAndSet(true, false);
499  }
500  } else {
501  Logger.d(TAG, "Event store not instantiated.");
502  isRunning.compareAndSet(true, false);
503  }
504  }

515  @SuppressWarnings("all")
516  protected LinkedList<RequestResult> performAsyncEmit(LinkedList<ReadyRequest> requests) {
517  LinkedList<RequestResult> results = new LinkedList<>();
518  LinkedList<Future> futures = new LinkedList<>();
519 
520  // Start all requests in the ThreadPool
521  for (ReadyRequest request : requests) {
522  futures.add(Executor.futureCallable(getRequestCallable(request.getRequest())));
523  }
524 
525  Logger.d(TAG, "Request Futures: %s", futures.size());
526 
527  // Get results of futures
528  // - Wait up to emitTimeout seconds for the request
529  for (int i = 0; i < futures.size(); i++) {
530  int code = -1;
531 
532  try {
533  code = (int) futures.get(i).get(emitTimeout, TimeUnit.SECONDS);
534  } catch (InterruptedException ie) {
535  Logger.e(TAG, "Request Future was interrupted: %s", ie.getMessage());
536  } catch (ExecutionException ee) {
537  Logger.e(TAG, "Request Future failed: %s", ee.getMessage());
538  } catch (TimeoutException te) {
539  Logger.e(TAG, "Request Future had a timeout: %s", te.getMessage());
540  }
541 
542  if (requests.get(i).isOversize()) {
543  results.add(new RequestResult(true, requests.get(i).getEventIds()));
544  } else {
545  results.add(new RequestResult(isSuccessfulSend(code), requests.get(i).getEventIds()));
546  }
547  }
548 
549  requests = null;
550  futures = null;
551 
552  return results;
553  }

605  protected LinkedList<ReadyRequest> buildRequests(EmittableEvents events) {
606 
607  int payloadCount = events.getEvents().size();
608  LinkedList<Long> eventIds = events.getEventIds();
609  LinkedList<ReadyRequest> requests = new LinkedList<>();
610 
611  if (httpMethod == HttpMethod.GET) {
612  for (int i = 0; i < payloadCount; i++) {
613 
614  // Get the eventId for this request
615  LinkedList<Long> reqEventId = new LinkedList<>();
616  reqEventId.add(eventIds.get(i));
617 
618  // Build and store the request
619  Payload payload = events.getEvents().get(i);
620  boolean oversize = payload.getByteSize() + POST_STM_BYTES > byteLimitGet;
621  Request request = requestBuilderGet(payload);
622  requests.add(new ReadyRequest(oversize, request, reqEventId));
623  }
624  } else {
625  for (int i = 0; i < payloadCount; i += bufferOption.getCode()) {
626 
627  LinkedList<Long> reqEventIds = new LinkedList<>();
628  ArrayList<Payload> postPayloadMaps = new ArrayList<>();
629  long totalByteSize = 0;
630 
631  for (int j = i; j < (i + bufferOption.getCode()) && j < payloadCount; j++) {
632  Payload payload = events.getEvents().get(j);
633  long payloadByteSize = payload.getByteSize() + POST_STM_BYTES;
634 
635  if ((payloadByteSize + POST_WRAPPER_BYTES) > byteLimitPost) {
636  ArrayList<Payload> singlePayloadMap = new ArrayList<>();
637  LinkedList<Long> reqEventId = new LinkedList<>();
638 
639  // Build and store the request
640  singlePayloadMap.add(payload);
641  reqEventId.add(eventIds.get(j));
642  Request request = requestBuilderPost(singlePayloadMap);
643  requests.add(new ReadyRequest(true, request, reqEventId));
644  }
645  else if ((totalByteSize + payloadByteSize + POST_WRAPPER_BYTES +
646  (postPayloadMaps.size() -1)) > byteLimitPost) {
647  Request request = requestBuilderPost(postPayloadMaps);
648  requests.add(new ReadyRequest(false, request, reqEventIds));
649 
650  // Clear collections and build a new POST
651  postPayloadMaps = new ArrayList<>();
652  reqEventIds = new LinkedList<>();
653 
654  // Build and store the request
655  postPayloadMaps.add(payload);
656  reqEventIds.add(eventIds.get(j));
657  totalByteSize = payloadByteSize;
658  } else {
659  totalByteSize += payloadByteSize;
660  postPayloadMaps.add(payload);
661  reqEventIds.add(eventIds.get(j));
662  }
663  }
664 
665  // Check if all payloads have been processed
666  if (!postPayloadMaps.isEmpty()) {
667  Request request = requestBuilderPost(postPayloadMaps);
668  requests.add(new ReadyRequest(false, request, reqEventIds));
669  }
670  }
671  }
672  return requests;
673  }
