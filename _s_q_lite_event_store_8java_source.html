184  @NonNull
185  private List<Map<String, Object>> queryDatabase(String query, String orderBy) {
186  List<Map<String, Object>> res = new ArrayList<>();
187  if (isDatabaseOpen()) {
188  Cursor cursor = null;
189  try {
190  cursor = database.query(EventStoreHelper.TABLE_EVENTS, allColumns, query, null, null, null, orderBy);
191  cursor.moveToFirst();
192  while (!cursor.isAfterLast()) {
193  Map<String, Object> eventMetadata = new HashMap<>();
194  eventMetadata.put(EventStoreHelper.METADATA_ID, cursor.getLong(0));
195  eventMetadata.put(EventStoreHelper.METADATA_EVENT_DATA,
196  Util.deserializer(cursor.getBlob(1)));
197  eventMetadata.put(EventStoreHelper.METADATA_DATE_CREATED, cursor.getString(2));
198  cursor.moveToNext();
199  res.add(eventMetadata);
200  }
201  } finally {
202  if (cursor != null) {
203  cursor.close();
204  }
205  }
206  }
207  return res;
208  }

232  @NonNull
233  @Override
234  public List<EmitterEvent> getEmittableEvents(int queryLimit) {
235  if (!isDatabaseOpen()) {
236  return Collections.emptyList();
237  }
238  insertWaitingEventsIfReady();
239 
240  ArrayList<EmitterEvent> events = new ArrayList<>();
241 
242  // FIFO Pattern for sending events
243  for (Map<String, Object> eventMetadata : getDescEventsInRange(queryLimit)) {
244 
245  // Create a TrackerPayload for each event
246  TrackerPayload payload = new TrackerPayload();
247  Map<String, Object> eventData = (Map<String, Object>)
248  eventMetadata.get(EventStoreHelper.METADATA_EVENT_DATA);
249  payload.addMap(eventData);
250 
251  // Create EmitterEvent
252  Long eventId = (Long) eventMetadata.get(EventStoreHelper.METADATA_ID);
253  if (eventId == null) {
254  Logger.e(TAG, "Unable to get ID of an event extracted from the database.");
255  continue;
256  }
257  EmitterEvent event = new EmitterEvent(payload, eventId);
258  events.add(event);
259  }
260  return events;
261  }
