93 public static long getUTF8Length(@NonNull String s) {
94 long len = 0;
95 for (int i = 0; i < s.length(); i++) {
96 char code = s.charAt(i);
97 if (code <= 0x7f) {
98 len += 1;
99 } else if (code <= 0x7ff) {
100 len += 2;
101 } else if (code >= 0xd800 && code <= 0xdfff) {
102 // Surrogate pair: These take 4 bytes in UTF-8 and 2 chars in UCS-2
103 // (Assume next char is the other [valid] half and just skip it)
104 len += 4; i++;
105 } else if (code < 0xffff) {
106 len += 3;
107 } else {
108 len += 4;
109 }
110 }
111 return len;
112 }

146 @NonNull
147 public static String joinLongList(@NonNull List<Long> list) {
148 StringBuilder s = new StringBuilder();
149
150 for (int i = 0; i < list.size(); i++) {
151 Long longVal = list.get(i);
152 if (longVal != null) {
153 s.append(list.get(i));
154 if (i < list.size() - 1) {
155 s.append(",");
156 }
157 }
158 }
159
160 if (s.toString().endsWith(",")) {
161 s = new StringBuilder(s.substring(0, s.length() - 1));
162 }
163
164 return s.toString();
165 }

175 @Nullable
176 public static SelfDescribingJson getGeoLocationContext(@NonNull Context context) {
177 Location location = getLastKnownLocation(context);
178
179 if (location != null) {
180 Map<String, Object> pairs = new HashMap<>();
181 addToMap(Parameters.LATITUDE, location.getLatitude(), pairs);
182 addToMap(Parameters.LONGITUDE, location.getLongitude(), pairs);
183 addToMap(Parameters.ALTITUDE, location.getAltitude(), pairs);
184 addToMap(Parameters.LATLONG_ACCURACY, location.getAccuracy(), pairs);
185 addToMap(Parameters.SPEED, location.getSpeed(), pairs);
186 addToMap(Parameters.BEARING, location.getBearing(), pairs);
187 addToMap(Parameters.GEO_TIMESTAMP, System.currentTimeMillis(), pairs);
188
189 if (mapHasKeys(pairs, Parameters.LATITUDE, Parameters.LONGITUDE)) {
190 return new SelfDescribingJson(
191 TrackerConstants.GEOLOCATION_SCHEMA, pairs
192 );
193 }
194 }
195 return null;
196 }

205 @SuppressLint("MissingPermission") // Suppressed as it's caught by SecurityException catch block.
206 @Nullable
207 public static Location getLastKnownLocation(@NonNull Context context) {
208 LocationManager locationManager = (LocationManager) context.getSystemService(Context.LOCATION_SERVICE);
209 Location location = null;
210
211 try {
212 String locationProvider = null;
213 if (locationManager.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
214 locationProvider = LocationManager.GPS_PROVIDER;
215 } else if (locationManager.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
216 locationProvider = LocationManager.NETWORK_PROVIDER;
217 } else {
218 List<String> locationProviders = locationManager.getProviders(true);
219 if (locationProviders.size() > 0) {
220 locationProvider = locationProviders.get(0);
221 }
222 }
223
224 if (locationProvider != null && !locationProvider.equals("")) {
225 location = locationManager.getLastKnownLocation(locationProvider);
226 }
227 } catch (SecurityException ex) {
228 Logger.e(TAG, "Exception occurred when retrieving location: %s", ex.toString());
229 }
230
231 return location;
232 }

239 @Nullable
240 static public SelfDescribingJson getApplicationContext(@NonNull Context context) {
241 try {
242 PackageInfo pInfo = context.getPackageManager().getPackageInfo(context.getPackageName(), 0);
243 String versionName = pInfo.versionName;
244 String versionCode;
245 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
246 versionCode = String.valueOf(pInfo.getLongVersionCode());
247 } else {
248 versionCode = String.valueOf(pInfo.versionCode);
249 }
250 if (versionName != null) {
251 Map<String, Object> pairs = new HashMap<>();
252 addToMap(Parameters.APP_VERSION, versionName, pairs);
253 addToMap(Parameters.APP_BUILD, versionCode, pairs);
254 return new SelfDescribingJson(
255 TrackerConstants.SCHEMA_APPLICATION, pairs
256 );
257 }
258 } catch (PackageManager.NameNotFoundException e) {
259 Logger.e(TAG, "Failed to find application context: %s", e.getMessage());
260 }
261 return null;
262 }

309 public static byte[] serialize(@NonNull Map<String, String> map) {
310 byte[] newByteArray = null;
311 try {
312 ByteArrayOutputStream mem_out = new ByteArrayOutputStream();
313 ObjectOutputStream out = new ObjectOutputStream(mem_out);
314 out.writeObject(map);
315 out.close();
316 mem_out.close();
317 newByteArray = mem_out.toByteArray();
318 } catch (IOException e) {
319 e.printStackTrace();
320 }

333 public static Map<String, String> deserializer(@NonNull byte[] bytes) {
334 Map<String, String> newMap = null;
335 try {
336 ByteArrayInputStream mem_in = new ByteArrayInputStream(bytes);
337 ObjectInputStream in = new ObjectInputStream(mem_in);
338 Map<String, String> map = (HashMap<String, String>) in.readObject();
339 in.close();
340 mem_in.close();
341 newMap = map;
342 } catch (NullPointerException | ClassNotFoundException | IOException e) {
343 e.printStackTrace();
344 }
345 return newMap;
